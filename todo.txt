JavascriptAutoloader

$JavascriptAutoloader = new JavascriptAutoloader();
$JavascriptAutoloader
	->setHomeUrl('http://test/')
	->addDirectory($rootDir . 'scripts')
	->addDirectory($rootDir . 'scripts/modules, $recursively = true)
	->setCacheStorage($rootDir . 'cache')
	->setUseMinify()
	->compileToOneFile()
	->autoload();
